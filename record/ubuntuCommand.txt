docker 启动容器
    docker run -d --name nacosorderapi -p 8912:8912 zhangqstudy/nacosorderapi:latest
查看镜像内容
    docker export temp_container | tar -t
删除镜像
    docker rm temp_container

sudo docker-compose -f docker-compose.yml pull
sudo docker-compose -f docker-compose.yml up -d

# 停止并删除所有容器
docker-compose down
# 删除相关镜像（可选）
docker rmi nacos/nacos-server:latest nginx:latest mysql:5.7 zookeeper:latest apache/kafka:4.1.1 redis:latest
# 重新构建并启动
docker-compose up -d
# 重新加载配置而不中断现有连接
docker exec -it <nginx-container-name> nginx -s reload
# 重启 nginx 服务容器
docker-compose restart nginx
# 特别检查Kafka和Zookeeper
docker ps | grep -E "(kafka|zookeeper)"
# 检查哪个进程占用了6379端口
sudo netstat -tlnp | grep :6379
# 或者
sudo lsof -i :6379
# 如果是Redis进程占用了端口，可以停止它
sudo kill <进程ID>

# 彻底清理包括数据卷
docker-compose down -v

列出所有本地镜像

    sudo systemctl deamon-reload
    sudo systemctl restart docker

查看docker状态
    sudo systemctl status docker

    docker ps


查看特定容器的详细信息
    docker inspect [容器名称或ID]
